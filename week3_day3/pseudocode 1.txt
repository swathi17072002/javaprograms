//pseudocode to print weekname by reading the week number

begin
   numeric week_num
   print("Enter Week number to find week day")
    input week_num
     if(week_num==1) {
            print("Monday")
        }
       else if (week_num==2) {
             print("Tuesday")
         }
       else if (week_num==3) {
             print("Wednesday")
         }
       else if (week_num==4) {
             print("Thursday")
         }
       else if (week_num==5) {
             print("Friday")
         }
       else if (week_num==6) {
             print("Saturday")
         }
       else if (week_num==7) {
             print("Sunday")
         }
       else {
           print("Invalid week number")
            }
 
end
